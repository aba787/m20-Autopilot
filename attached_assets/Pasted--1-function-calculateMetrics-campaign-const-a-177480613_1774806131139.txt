function applyRules(metrics, campaign) {
  if (metrics.acos > campaign.target_acos && campaign.clicks > 25) {
    return {
      action: "decrease_bid",
      reason: "ACOS too high"
    }
  }

  if (metrics.acos < campaign.target_acos * 0.7) {
    return {
      action: "scale",
      reason: "Good performance"
    }
  }

  return {
    action: "keep",
    reason: "Stable"
  }
}
