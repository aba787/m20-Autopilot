SQL
CREATE TABLE amazon_connections (
  id uuid DEFAULT uuid_generate_v4() PRIMARY KEY,
  user_id uuid REFERENCES auth.users(id) NOT NULL, -- لربط الحساب بمستخدم موقعك
  access_token TEXT NOT NULL,
  refresh_token TEXT NOT NULL,
  expires_at TIMESTAMP WITH TIME ZONE NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
الخطوة 2: إرسال المستخدم لأمازون (Backend)
في ملف الـ API الخاص بك (مثلاً amazon-connect.js):

JavaScript
// GET /api/amazon/connect
export default async function handler(req, res) {
  const CLIENT_ID = process.env.AMAZON_CLIENT_ID;
  const REDIRECT_URI = process.env.AMAZON_REDIRECT_URI;
  
  // الـ Scope يحدد وش مسموح لك تسوين (هنا إدارة الحملات)
  const scope = "advertising::campaign_management";
  
  const url = `https://www.amazon.com/ap/oa?client_id=${CLIENT_ID}&response_type=code&scope=${scope}&redirect_uri=${REDIRECT_URI}`;
  
  // نوجه المستخدم لصفحة تسجيل دخول أمازون
  res.redirect(url);
}
الخطوة 3: استقبال الـ Code وتبديله بالـ Tokens (Backend)
هذا الجزء هو "قلب" العملية في ملف amazon-callback.js:

JavaScript
// GET /api/amazon/callback
export default async function handler(req, res) {
  const { code } = req.query; // أمازون بيرسل الكود في الرابط
  
  if (!code) return res.status(400).send("No code provided");

  // 1. طلب الـ Tokens من أمازون باستخدام الكود
  const response = await fetch('https://api.amazon.com/auth/o2/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: new URLSearchParams({
      grant_type: 'authorization_code',
      code: code,
      client_id: process.env.AMAZON_CLIENT_ID,
      client_secret: process.env.AMAZON_CLIENT_SECRET,
      redirect_uri: process.env.AMAZON_REDIRECT_URI,
    }),
  });

  const data = await response.json();

  // 2. تخزين البيانات في Supabase
  // ملاحظة: expires_in غالباً يكون بالثواني (3600 ثانية = ساعة)
  const expiresAt = new Date(Date.now() + data.expires_in * 1000);

  const { error } = await supabase
    .from('amazon_connections')
    .insert([
      { 
        user_id: req.user.id, // ID المستخدم المسجل حالياً
        access_token: data.access_token, 
        refresh_token: data.refresh_token, 
        expires_at: expiresAt 
      }
    ]);

  if (error) return res.status(500).json({ error: error.message });

  // 3. إعادة توجيه المستخدم للـ Dashboard في موقعك
  res.redirect('/dashboard?status=success');
}
الخطوة 4: سحب الحملات (The Real Work)
الآن بما أن التوكن محفوظ، يمكنك استخدامه في أي وقت لسحب البيانات:

JavaScript
async function getAmazonCampaigns(userId) {
  // 1. نجيب التوكن من القاعدة
  const { data: connection } = await supabase
    .from('amazon_connections')
    .select('*')
    .eq('user_id', userId)
    .single();

  // 2. نرسل الطلب لأمازون
  const response = await fetch('https://advertising-api.amazon.com/v2/campaigns', {
    headers: {
      'Authorization': `Bearer ${connection.access_token}`,
      'Amazon-Advertising-API-ClientId': process.env.AMAZON_CLIENT_ID,
      'Content-Type': 'application/json'
    }
  });

  return await response.json();
}
